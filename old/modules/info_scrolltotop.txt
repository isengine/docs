Модуль вывода кнопки "в начало" со скроллингом страницы

Шаблон задается самостоятельно, после вызова модуля, работа в шаблоне идет с объектом $module

{
	"classes" : {
	// классы для элементов
	// значения: string
	// элементы: base, icon, text
	// строка с классами для базового элемента (по-умолчанию "scroll-to-top") или массив с указанием классов для отдельных объектов
		
		"base" : "",
		// классы для базового элемента (по-умолчанию "scroll-to-top")
		
		"icon" : "",
		// классы для значка внутри базового элемента
		
		"text" : "",
		// классы для текста внутри базового элемента
		
	"api" : "vk",
	// значения: string: vk/instagram
	
	"lang" : "",
	// значения: string
	// параметры для строковой переменной {lang:...}
	// если хотите использовать строковую переменную, просто укажите здесь ее параметры
	
	"text" : "",
	// значения: string
	// выводимый текст (в целях "защиты от дурака" не принимает строковые переменные)
	
	"distance" : 100,
	// значения: num, string
	// расстояние прокрутки, после которого появляется значок
	// необязательный, по-умолчанию: 100
	
	"speed" : 500,
	// значения: num, string
	// скорость прокрутки страницы вверх
	// необязательный, по-умолчанию: 500
	
	"element" : "a"
	// значения: string
	// базовый элемент
	// допустимые значения: a, button, div, p, span
	// необязательный, по-умолчанию: a (ссылка)
	
}

также text может содержать локализации, например:
	
	"text" : {
		"en" : "hint", // надпись на английском
		"ru" : "хинт" // надпись на русском
	}
