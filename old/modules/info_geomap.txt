если значков нет, то на карте по-умолчанию ставится один значок в самом центре
marks также может быть строкой типа "pin.png", тогда в центре карты будет один значок с указанной картинкой

картинка должна обязательно быть в заданной папке с медиа, например:
	папка - media
	для image = pin.png, ссылка будет /media/pin.png
	для image = map_icon/pin.png, ссылка будет /media/map_icon/pin.png

значения в массиве marks:
	
	"image" : "pin.png",
	// ссылка на рисунок маркера
	// значения: string
	
	"coordinates" : ["44.049700", "42.866577"],
	// координаты маркера
	// значения: array num/string
	// если не указаны, то будут взяты координаты центра карты
	
	"hint" : "hint",
	// надпись, появляющаяся при наведении на маркер
	// значения: string
	
	"header" : "заголовок",
	// заголовок всплывающего окна при щелчке по маркеру
	// значения: string
	
	"content" : "<p>содержимое</p>",
	// содержимое всплывающего окна при щелчке по маркеру
	// значения: string
	
	"footer" : "подвал или подпись",
	// подвал всплывающего окна при щелчке по маркеру
	// значения: string
	
	// для google header, content и footer будут объединены в один общий контент со следующими классами:
	// map_<?= $module -> param; ?>_header
 	// map_<?= $module -> param; ?>_content
	// map_<?= $module -> param; ?>_footer
	
	"autoopen" : true/false
	// если вы хотите, чтобы всплывающее окно открывалось автоматически при загрузке карты
	
	// следующие параметры ТОЛЬКО (!) для яндекс карт
	
	"caption" : "",
	// текст возле иконки
	// значения: string
	
	"preset" : "",
	// вид иконки
	// значения: string
	// доступные значения: dotIcon, circleIcon, circleDotIcon и т.д. (из списка https://tech.yandex.ru/maps/doc/jsapi/2.1/ref/reference/option.presetStorage-docpage/)
	
	"color" : ""
	// цвет иконки
	// значения: string
	// название цвета или #rrggbb

также hint, header, content и footer могут содержать локализации, например:
	
	"hint" : {
		"en" : "hint", // надпись на английском
		"ru" : "хинт" // надпись на русском
	}
